### A/B Test Store (`stores/abtest.js`)

```javascript
// src/stores/abtest.js
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { DEBUG_MODE } from '@/config/settings.js'

export const useABTestStore = defineStore('abtest', () => {
  // –ö–æ–Ω—Ñ–∏–≥—É—Ä–∞—Ü–∏—è —ç–∫—Å–ø–µ—Ä–∏–º–µ–Ω—Ç–æ–≤
  const experiments = ref({
    onboarding_v2: {
      name: 'Onboarding A/B Test',
      variants: ['fast', 'deep'],  // A = fast, B = deep
      weights: [50, 50],           // 50/50 —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ
      enabled: true
    }
  })

  // –ù–∞–∑–Ω–∞—á–µ–Ω–Ω—ã–µ –≤–∞—Ä–∏–∞–Ω—Ç—ã –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—é (–∫—ç—à)
  const userVariants = ref({})

  /**
   * –ü–æ–ª—É—á–∏—Ç—å –∏–ª–∏ –Ω–∞–∑–Ω–∞—á–∏—Ç—å –≤–∞—Ä–∏–∞–Ω—Ç –¥–ª—è —ç–∫—Å–ø–µ—Ä–∏–º–µ–Ω—Ç–∞
   * –°–æ—Ö—Ä–∞–Ω—è–µ—Ç—Å—è –≤ localStorage –¥–ª—è –∫–æ–Ω—Å–∏—Å—Ç–µ–Ω—Ç–Ω–æ—Å—Ç–∏
   */
  function getVariant(experimentId) {
    // 1. –ü—Ä–æ–≤–µ—Ä—è–µ–º localStorage
    const storageKey = `ab_${experimentId}`
    const stored = localStorage.getItem(storageKey)

    if (stored && userVariants.value[experimentId] === undefined) {
      userVariants.value[experimentId] = stored
      return stored
    }

    // 2. –ü—Ä–æ–≤–µ—Ä—è–µ–º –∫—ç—à
    if (userVariants.value[experimentId]) {
      return userVariants.value[experimentId]
    }

    // 3. –ù–∞–∑–Ω–∞—á–∞–µ–º –Ω–æ–≤—ã–π –≤–∞—Ä–∏–∞–Ω—Ç
    const experiment = experiments.value[experimentId]
    if (!experiment || !experiment.enabled) {
      return null
    }

    const variant = assignVariant(experiment.variants, experiment.weights)
    userVariants.value[experimentId] = variant
    localStorage.setItem(storageKey, variant)

    if (DEBUG_MODE) {
      console.log(`[ABTest] Assigned variant "${variant}" for "${experimentId}"`)
    }

    return variant
  }

  /**
   * –°–ª—É—á–∞–π–Ω–æ–µ –Ω–∞–∑–Ω–∞—á–µ–Ω–∏–µ —Å —É—á—ë—Ç–æ–º –≤–µ—Å–æ–≤
   */
  function assignVariant(variants, weights) {
    const totalWeight = weights.reduce((sum, w) => sum + w, 0)
    const random = Math.random() * totalWeight

    let cumulative = 0
    for (let i = 0; i < variants.length; i++) {
      cumulative += weights[i]
      if (random < cumulative) {
        return variants[i]
      }
    }
    return variants[0]
  }

  /**
   * –ü—Ä–∏–Ω—É–¥–∏—Ç–µ–ª—å–Ω–æ —É—Å—Ç–∞–Ω–æ–≤–∏—Ç—å –≤–∞—Ä–∏–∞–Ω—Ç (–¥–ª—è —Ç–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏—è)
   */
  function forceVariant(experimentId, variant) {
    userVariants.value[experimentId] = variant
    localStorage.setItem(`ab_${experimentId}`, variant)

    if (DEBUG_MODE) {
      console.log(`[ABTest] Forced variant "${variant}" for "${experimentId}"`)
    }
  }

  /**
   * –°–±—Ä–æ—Å–∏—Ç—å –≤–∞—Ä–∏–∞–Ω—Ç (–¥–ª—è —Ç–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏—è)
   */
  function resetVariant(experimentId) {
    delete userVariants.value[experimentId]
    localStorage.removeItem(`ab_${experimentId}`)

    if (DEBUG_MODE) {
      console.log(`[ABTest] Reset variant for "${experimentId}"`)
    }
  }

  // Computed –¥–ª—è —É–¥–æ–±—Å—Ç–≤–∞
  const onboardingVariant = computed(() => getVariant('onboarding_v2'))

  return {
    experiments,
    userVariants,
    getVariant,
    forceVariant,
    resetVariant,
    onboardingVariant
  }
})
```

### OnboardingRouter.vue

```vue
<!-- src/components/onboarding/OnboardingRouter.vue -->
<template>
  <component
    :is="currentComponent"
    @complete="handleComplete"
    @track="handleTrack"
  />
</template>

<script setup>
import { computed, onMounted, ref } from 'vue'
import { useRouter, useRoute } from 'vue-router'
import { useABTestStore } from '@/stores/abtest'
import { DEBUG_MODE } from '@/config/settings.js'
import * as api from '@/services/api.js'

// Lazy load –∫–æ–º–ø–æ–Ω–µ–Ω—Ç–æ–≤
import OnboardingFast from './OnboardingFast.vue'
import OnboardingDeep from './OnboardingDeep.vue'
import OnboardingAI from '../OnboardingAI.vue'

const router = useRouter()
const route = useRoute()
const abStore = useABTestStore()
const startTime = ref(Date.now())

// –ü—Ä–æ–≤–µ—Ä–∫–∞ URL –ø–∞—Ä–∞–º–µ—Ç—Ä–∞ –¥–ª—è –ø—Ä–∏–Ω—É–¥–∏—Ç–µ–ª—å–Ω–æ–≥–æ –≤–∞—Ä–∏–∞–Ω—Ç–∞
onMounted(() => {
  const forceVariant = route.query.ab_force
  if (forceVariant && ['fast', 'deep'].includes(forceVariant)) {
    abStore.forceVariant('onboarding_v2', forceVariant)
  }

  // –¢—Ä–µ–∫–∞–µ–º —Å—Ç–∞—Ä—Ç
  trackEvent('onboarding_started', {
    variant: abStore.onboardingVariant,
    timestamp: new Date().toISOString()
  })
})

const currentComponent = computed(() => {
  const variant = abStore.onboardingVariant

  if (DEBUG_MODE) {
    console.log('[OnboardingRouter] Current variant:', variant)
  }

  switch (variant) {
    case 'fast':
      return OnboardingFast
    case 'deep':
      return OnboardingDeep
    default:
      return OnboardingAI  // Fallback
  }
})

function handleComplete(data) {
  const duration = Math.round((Date.now() - startTime.value) / 1000)

  trackEvent('onboarding_completed', {
    variant: abStore.onboardingVariant,
    duration_seconds: duration,
    goals_accepted: data.goalsAccepted || 0,
    steps_completed: data.stepsCompleted || 0
  })
}

function handleTrack(eventName, eventData = {}) {
  trackEvent(eventName, {
    variant: abStore.onboardingVariant,
    ...eventData
  })
}

async function trackEvent(eventName, data) {
  if (DEBUG_MODE) {
    console.log(`[OnboardingRouter] Track: ${eventName}`, data)
  }

  try {
    await api.trackABEvent({
      experiment: 'onboarding_v2',
      event: eventName,
      variant: abStore.onboardingVariant,
      data,
      timestamp: new Date().toISOString()
    })
  } catch (error) {
    if (DEBUG_MODE) {
      console.error('[OnboardingRouter] Failed to track event:', error)
    }
  }
}
</script>
```

```javascript
/**
 * –†–∞—Å—Å—á–∏—Ç–∞—Ç—å –ø—Ä–æ–≥–Ω–æ–∑ —É–ª—É—á—à–µ–Ω–∏—è
 * @param {Array} spheres - –ú–∞—Å—Å–∏–≤ —Å—Ñ–µ—Ä —Å –æ—Ü–µ–Ω–∫–∞–º–∏
 * @param {string} timeCommitment - –í—ã–±—Ä–∞–Ω–Ω–æ–µ –≤—Ä–µ–º—è (15_mins, 30_mins, etc)
 * @returns {Object} - –ü—Ä–æ–≥–Ω–æ–∑
 */
export function calculateProgress(spheres, timeCommitment) {
  const currentAvg = spheres.reduce((sum, s) => sum + s.score, 0) / spheres.length

  // –ö–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç —Ä–æ—Å—Ç–∞ –≤ –∑–∞–≤–∏—Å–∏–º–æ—Å—Ç–∏ –æ—Ç –≤—Ä–µ–º–µ–Ω–∏

    '15_mins': 0.15,  // 15% –∑–∞ 90 –¥–Ω–µ–π
    '30_mins': 0.25,  // 25% –∑–∞ 90 –¥–Ω–µ–π
    '1_hour': 0.40,   // 40% –∑–∞ 90 –¥–Ω–µ–π
    'something': 0.10 // 10% –∑–∞ 90 –¥–Ω–µ–π
  }

  const rate = growthRates[timeCommitment] || 0.15
  const potentialAvg = Math.min(10, currentAvg + (10 - currentAvg) * rate)

  // –î–∞—Ç–∞ –¥–æ—Å—Ç–∏–∂–µ–Ω–∏—è "–±–∞–ª–∞–Ω—Å–∞" (7.0 —Å—Ä–µ–¥–Ω–µ–≥–æ)
  const targetAvg = 7.0
  const daysNeeded = currentAvg >= targetAvg
    ? 0
    : Math.round(90 * (targetAvg - currentAvg) / (potentialAvg - currentAvg))

  const targetDate = new Date()
  targetDate.setDate(targetDate.getDate() + daysNeeded)

  return {
    currentAvg: currentAvg.toFixed(1),
    potentialAvg: potentialAvg.toFixed(1),
    improvementPercent: Math.round((potentialAvg - currentAvg) / currentAvg * 100),
    daysToBalance: daysNeeded,
    targetDate: targetDate.toLocaleDateString('ru-RU', {
      day: 'numeric',
      month: 'long',
      year: 'numeric'
    }),
    weakestSphere: spheres.reduce((min, s) => s.score < min.score ? s : min),
    strongestSphere: spheres.reduce((max, s) => s.score > max.score ? s : max)
  }
}
```
